Aquí tienes la sentencia para agregar INCUMPLIMIENTO al ENUM de estado:

ALTER TABLE tbl_pendientes
  MODIFY COLUMN estado ENUM('ABIERTA','CERRADA','INCUMPLIMIENTO') NOT NULL;

ALTER TABLE tbl_pendientes
  MODIFY COLUMN estado ENUM('ABIERTA','CERRADA','INCUMPLIMIENTO') NOT NULL DEFAULT 'ABIERTA';
